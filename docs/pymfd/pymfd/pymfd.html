 32class _InstantiationTrackerMixin:
 33    """Mixin class to track where a component was instantiated."""
 34
 35    def __init__(self):
 36        # Get the first relevant frame outside of this class
 37        for frame_info in inspect.stack():
 38            filename = frame_info.filename
 39            if "site-packages" in filename or filename == __file__:
 40                continue
 41            self._instantiation_path = Path(filename).resolve()
 42            break
 43
 44    @property
 45    def instantiation_dir(self) -> Path:
 46        """Return (directory, filename_stem) of the file that instantiated the component, if it's a Device or Component."""
 47        class_name = type(self).__name__
 48
 49        if class_name in {"Device", "Component"}:
 50            return self._instantiation_path.parent
 51
 52        # Fallback: use where the class is defined
 53        module_name = self.__class__.__module__
 54        module = sys.modules.get(module_name) or importlib.import_module(module_name)
 55        path = Path(module.__file__).resolve()
 56        return path.parent
 57
 58    @property
 59    def instantiating_file_stem(self) -> str:
 60        """Return (directory, filename_stem) of the file that instantiated the component, if it's a Device or Component."""
 61        class_name = type(self).__name__
 62
 63        if class_name in {"Device", "Component"}:
 64            return self._instantiation_path.stem
 65
 66        # Fallback: use where the class is defined
 67        module_name = self.__class__.__module__
 68        module = sys.modules.get(module_name) or importlib.import_module(module_name)
 69        path = Path(module.__file__).resolve()
 70        return path.stem

 73class Port(_InstantiationTrackerMixin):
 74    """
 75    ###### Class representing a port in a microfluidic device.
 76    ###### Ports are used to connect components and define their interaction with the environment.
 77    ###### Each port has a type (IN, OUT, INOUT), a position in 3D space, a size, and a surface normal.
 78    ###### The surface normal defines the direction in which the port is oriented.
 79    """
 80
 81    class PortType(Enum):
 82        """Enumeration for port types."""
 83
 84        IN = 1
 85        OUT = 2
 86        INOUT = 3
 87
 88    class SurfaceNormal(Enum):
 89        """Enumeration for surface normals."""
 90
 91        POS_X = 1
 92        POS_Y = 2
 93        POS_Z = 3
 94        NEG_X = 4
 95        NEG_Y = 5
 96        NEG_Z = 6
 97
 98    _vector_map = {
 99        SurfaceNormal.POS_X: (1, 0, 0),
100        SurfaceNormal.POS_Y: (0, 1, 0),
101        SurfaceNormal.POS_Z: (0, 0, 1),
102        SurfaceNormal.NEG_X: (-1, 0, 0),
103        SurfaceNormal.NEG_Y: (0, -1, 0),
104        SurfaceNormal.NEG_Z: (0, 0, -1),
105    }
106
107    def __init__(
108        self,
109        _type: PortType,
110        position: tuple[int, int, int],
111        size: tuple[int, int, int],
112        surface_normal: SurfaceNormal,
113    ):
114        """
115        ###### Initialize a port.
116
117        ###### Parameters:
118        - _type (PortType): The type of the port (IN, OUT, INOUT).
119        - position (tuple[int, int, int]): The position of the port in 3D space.
120        - size (tuple[int, int, int]): The size of the port.
121        - surface_normal (SurfaceNormal): The surface normal of the port, defining its orientation.
122        """
123        super().__init__()
124        self._parent = None
125        self._name = None
126        self._type = _type
127        self._position = position
128        self._size = size
129        self._surface_normal = surface_normal
130
131    def get_name(self) -> str:
132        """Get the name of the port, including parent name."""
133        if self._name is None:
134            raise ValueError(f"Port has not been named")
135        else:
136            return f"{self._parent._name}_{self._name}"
137
138    def get_fully_qualified_name(self) -> str:
139        """Get the fully qualified name of the port, including all parent components names."""
140        if self._name is None:
141            raise ValueError(f"Port has not been named")
142        name = self._name
143        parent = self._parent
144        while parent is not None:
145            if parent._name is not None:
146                name = f"{parent._name}.{name}"
147                parent = parent._parent
148            else:
149                name = f"{parent.instantiating_file_stem}.{name}"
150                return name
151        return name
152
153    def to_vector(self) -> tuple[int, int, int]:
154        """Convert the surface normal to a vector."""
155        try:
156            return self._vector_map[self._surface_normal]
157        except KeyError:
158            raise ValueError(f"Unsupported surface normal: {self._surface_normal}")
159
160    def get_bounding_box(self) -> tuple[int, int, int, int, int, int]:
161        """
162        Get the bounding box of the port.
163        The bounding box is defined by the position and size of the port,
164        adjusted based on the surface normal direction.
165
166        Returns:
167        - A tuple of six integers representing the bounding box coordinates:
168        (min_x, min_y, min_z, max_x, max_y, max_z)
169        """
170        dx, dy, dz = self._vector_map[self._surface_normal]
171        pos = list(self._position)
172        size = self._size
173
174        # For negative directions, shift start back by size
175        if dx < 0:
176            pos[0] -= size[0]
177        if dy < 0:
178            pos[1] -= size[1]
179        if dz < 0:
180            pos[2] -= size[2]
181
182        return (
183            pos[0],
184            pos[1],
185            pos[2],
186            pos[0] + size[0],
187            pos[1] + size[1],
188            pos[2] + size[2],
189        )
190
191    def get_origin(self):
192        """Get the origin of the port, which is the minimum corner of its bounding box."""
193        return self.get_bounding_box()[0:3]
194
195    def get_color(self):
196        """
197        ###### Get the color of the port based on its type.
198
199        ###### The color is determined as follows:
200        - IN ports are green
201        - OUT ports are red
202        - INOUT ports are blue
203        - If the type is not recognized, it defaults to white.
204
205        ###### Returns:
206        - Color: The color of the port.
207        """
208        if self._type == Port.PortType.IN:
209            return Color.from_name("green", 255)  # Green
210        elif self._type == Port.PortType.OUT:
211            return Color.from_name("red", 255)  # Red
212        elif self._type == Port.PortType.INOUT:
213            return Color.from_name("blue", 255)  # Blue
214        else:
215            return Color.from_name("white", 255)  # White

291    def _validate_name(self, name):
292        """
293        ###### Validate the name for a new label, port, shape, or subcomponent.
294        ###### Raises:
295        - ValueError: If the name already exists in the component or is not a valid Python identifier.
296        """
297        for l in self.labels:
298            if l == name:
299                raise ValueError(
300                    f"Label with name '{name}' already exists in component {self._name}"
301                )
302        for p in self.ports:
303            if p._name == name:
304                raise ValueError(
305                    f"Port with name '{name}' already exists in component {self._name}"
306                )
307        for s in self.shapes:
308            if s._name == name:
309                raise ValueError(
310                    f"Shape with name '{name}' already exists in component {self._name}"
311                )
312        for c in self.subcomponents:
313            if c._name == name:
314                raise ValueError(
315                    f"Commponent with name '{name}' already exists in component {self._name}"
316                )
317        if not name.isidentifier():
318            raise ValueError(
319                f"Name '{name}' is not a valid Python identifier (e.g. no spaces, starts with letter, etc.)"
320            )
321        if hasattr(self, name):
322            raise ValueError(f"Name '{name}' conflicts with existing attributes")

612    def rotate(self, rotation: int, in_place: bool = False) -> Component:
613        """
614        ###### Rotate the component around the Z axis by a given angle.
615        ###### Parameters:
616        - rotation (int): The angle in degrees to rotate the component. Must be a multiple of 90.
617        - in_place (bool): If True, the component is rotated in place. Default is False.
618        ###### Returns:
619        - self: The rotated component.
620        """
621        if rotation % 90 != 0:
622            raise ValueError("Rotation must be a multiple of 90 degrees")
623
624        if in_place:
625            original_position = self._position
626            # Translate the component to position for in-place rotation
627            self.translate(
628                (-self._position[0], -self._position[1], -self._position[2]),
629                set_origin=False,
630            )
631
632        for component in self.subcomponents:
633            component.rotate(rotation)
634
635        for shape in self.shapes:

739        self, mirror_x: bool = False, mirror_y: bool = False, in_place: bool = False
740    ) -> Component:
741        """
742        ###### Mirror the component along the X and/or Y axes.
743        ###### Parameters:
744        - mirror_x (bool): If True, mirrors the component along the X axis. Default is False.
745        - mirror_y (bool): If True, mirrors the component along the Y axis. Default is False.
746        - in_place (bool): If True, performs the mirroring in place. Default is False.
747        ###### Returns:
748        - self: The mirrored component.
749        """
750        if not mirror_x and not mirror_y:
751            return self  # No mirroring requested
752
753        if mirror_x and mirror_y:
754            return self.rotate(180, in_place=in_place)
755
756        if in_place:
757            original_position = self._position
758            # Translate the component to position for in-place mirroring
759            self.translate(
760                (-self._position[0], -self._position[1], -self._position[2]),
761                set_origin=False,
762            )
763
764        for component in self.subcomponents:
765            component.mirror(mirror_x, mirror_y)
766
767        for shape in self.shapes:
768            shape.mirror((mirror_x, mirror_y, False))
769
770        # Surface normal flips
771        mirror_vector_map = {
772            "x": {
773                Port.SurfaceNormal.POS_X: Port.SurfaceNormal.NEG_X,
774                Port.SurfaceNormal.NEG_X: Port.SurfaceNormal.POS_X,
775            },
776            "y": {
777                Port.SurfaceNormal.POS_Y: Port.SurfaceNormal.NEG_Y,
778                Port.SurfaceNormal.NEG_Y: Port.SurfaceNormal.POS_Y,
779            },
780        }
781
782        for port in self.ports:
783            x, y, z = port._position
784            sx, sy = port._size[0], port._size[1]
785
786            if mirror_x:
787                x = -x - sx
788                # If pointing in +X or -X, correct for sticking out
789                if port._surface_normal == Port.SurfaceNormal.POS_X:
790                    x += sx
791                elif port._surface_normal == Port.SurfaceNormal.NEG_X:
792                    x += sx
793                port._surface_normal = mirror_vector_map["x"].get(
794                    port._surface_normal, port._surface_normal
795                )
796
797            if mirror_y:
798                y = -y - sy
799                # If pointing in +Y or -Y, correct for sticking out
800                if port._surface_normal == Port.SurfaceNormal.POS_Y:
801                    y += sy
802                elif port._surface_normal == Port.SurfaceNormal.NEG_Y:
803                    y += sy
804                port._surface_normal = mirror_vector_map["y"].get(
805                    port._surface_normal, port._surface_normal
806                )
807
808            port._position = (x, y, z)
809
810        if in_place:
811            # Translate the component so new negative-negative corner is at original position
812            if mirror_x and not mirror_y:
813                self.translate(
814                    (
815                        original_position[0] + self._size[0],
816                        original_position[1],
817                        original_position[2],
818                    ),
819                    set_origin=False,
820                )
821            elif not mirror_x and mirror_y:
822                self.translate(
823                    (
824                        original_position[0],
825                        original_position[1] + self._size[1],
826                        original_position[2],
827                    ),
828                    set_origin=False,
829                )
830
831        return self

613    def rotate(self, rotation: int, in_place: bool = False) -> Component:
614        """
615        ###### Rotate the component around the Z axis by a given angle.
616        ###### Parameters:
617        - rotation (int): The angle in degrees to rotate the component. Must be a multiple of 90.
618        - in_place (bool): If True, the component is rotated in place. Default is False.
619        ###### Returns:
620        - self: The rotated component.
621        """
622        if rotation % 90 != 0:
623            raise ValueError("Rotation must be a multiple of 90 degrees")
624
625        if in_place:
626            original_position = self._position
627            # Translate the component to position for in-place rotation
628            self.translate(
629                (-self._position[0], -self._position[1], -self._position[2]),
630                set_origin=False,
631            )
632
633        for component in self.subcomponents:
634            component.rotate(rotation)
635
636        for shape in self.shapes:
637            shape.rotate((0, 0, rotation))
638
639        rot = rotation % 360
640
641        # Mapping for 90-degree rotation steps around Z axis
642        vector_rotation_map = {
643            90: {
644                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.POS_Y, (-1, 0)),
645                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.NEG_X, (0, 0)),
646                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.NEG_Y, (-1, 0)),
647                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.POS_X, (0, 0)),
648            },
649            180: {
650                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.NEG_X, (0, -1)),
651                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.NEG_Y, (-1, 0)),
652                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.POS_X, (0, -1)),
653                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.POS_Y, (-1, 0)),
654            },
655            270: {
656                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.NEG_Y, (0, 0)),
657                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.POS_X, (0, -1)),
658                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.POS_Y, (0, 0)),
659                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.NEG_X, (0, -1)),
660            },
661        }
662
663        for port in self.ports:
664            x, y, z = port._position
665            size_x, size_y = port._size[0], port._size[1]
666
667            if rot == 90:
668                port._position = (-y, x, z)
669            elif rot == 180:
670                port._position = (-x, -y, z)
671            elif rot == 270:
672                port._position = (y, -x, z)
673
674            if port._surface_normal in vector_rotation_map.get(rot, {}):
675                new_vector, (dx, dy) = vector_rotation_map[rot][port._surface_normal]
676                port._position = (
677                    port._position[0] + dx * port._size[0],
678                    port._position[1] + dy * port._size[1],
679                    port._position[2],
680                )
681                port._surface_normal = new_vector
682
683            # Adjust Z ports if needed
684            if port._surface_normal in (
685                Port.SurfaceNormal.POS_Z,
686                Port.SurfaceNormal.NEG_Z,
687            ):
688                if rot == 90:
689                    port._position = (
690                        port._position[0] - port._size[0],
691                        port._position[1],
692                        port._position[2],
693                    )
694                elif rot == 180:
695                    port._position = (
696                        port._position[0] - port._size[0],
697                        port._position[1] - port._size[1],
698                        port._position[2],
699                    )
700                elif rot == 270:
701                    port._position = (
702                        port._position[0],
703                        port._position[1] - port._size[1],
704                        port._position[2],
705                    )
706
707        if in_place:
708            # Translate the component so new negative-negative corner is at original position
709            if rot == 90:
710                self.translate(
711                    (
712                        original_position[0] + self._size[1],
713                        original_position[1],
714                        original_position[2],
715                    ),
716                    set_origin=False,
717                )
718            elif rot == 180:
719                self.translate(
720                    (
721                        original_position[0] + self._size[0],
722                        original_position[1] + self._size[1],
723                        original_position[2],
724                    ),
725                    set_origin=False,
726                )
727            elif rot == 270:
728                self.translate(
729                    (
730                        original_position[0],
731                        original_position[1] + self._size[0],
732                        original_position[2],
733                    ),
734                    set_origin=False,
735                )
736
737        return self

739    def mirror(
740        self, mirror_x: bool = False, mirror_y: bool = False, in_place: bool = False
741    ) -> Component:
742        """
743        ###### Mirror the component along the X and/or Y axes.
744        ###### Parameters:
745        - mirror_x (bool): If True, mirrors the component along the X axis. Default is False.
746        - mirror_y (bool): If True, mirrors the component along the Y axis. Default is False.
747        - in_place (bool): If True, performs the mirroring in place. Default is False.
748        ###### Returns:
749        - self: The mirrored component.
750        """
751        if not mirror_x and not mirror_y:
752            return self  # No mirroring requested
753
754        if mirror_x and mirror_y:
755            return self.rotate(180, in_place=in_place)
756
757        if in_place:
758            original_position = self._position
759            # Translate the component to position for in-place mirroring
760            self.translate(
761                (-self._position[0], -self._position[1], -self._position[2]),
762                set_origin=False,
763            )
764
765        for component in self.subcomponents:
766            component.mirror(mirror_x, mirror_y)
767
768        for shape in self.shapes:
769            shape.mirror((mirror_x, mirror_y, False))
770
771        # Surface normal flips
772        mirror_vector_map = {
773            "x": {
774                Port.SurfaceNormal.POS_X: Port.SurfaceNormal.NEG_X,
775                Port.SurfaceNormal.NEG_X: Port.SurfaceNormal.POS_X,
776            },
777            "y": {
778                Port.SurfaceNormal.POS_Y: Port.SurfaceNormal.NEG_Y,
779                Port.SurfaceNormal.NEG_Y: Port.SurfaceNormal.POS_Y,
780            },
781        }
782
783        for port in self.ports:
784            x, y, z = port._position
785            sx, sy = port._size[0], port._size[1]
786
787            if mirror_x:
788                x = -x - sx
789                # If pointing in +X or -X, correct for sticking out
790                if port._surface_normal == Port.SurfaceNormal.POS_X:
791                    x += sx
792                elif port._surface_normal == Port.SurfaceNormal.NEG_X:
793                    x += sx
794                port._surface_normal = mirror_vector_map["x"].get(
795                    port._surface_normal, port._surface_normal
796                )
797
798            if mirror_y:
799                y = -y - sy
800                # If pointing in +Y or -Y, correct for sticking out
801                if port._surface_normal == Port.SurfaceNormal.POS_Y:
802                    y += sy
803                elif port._surface_normal == Port.SurfaceNormal.NEG_Y:
804                    y += sy
805                port._surface_normal = mirror_vector_map["y"].get(
806                    port._surface_normal, port._surface_normal
807                )
808
809            port._position = (x, y, z)
810
811        if in_place:
812            # Translate the component so new negative-negative corner is at original position
813            if mirror_x and not mirror_y:
814                self.translate(
815                    (
816                        original_position[0] + self._size[0],
817                        original_position[1],
818                        original_position[2],
819                    ),
820                    set_origin=False,
821                )
822            elif not mirror_x and mirror_y:
823                self.translate(
824                    (
825                        original_position[0],
826                        original_position[1] + self._size[1],
827                        original_position[2],
828                    ),
829                    set_origin=False,
830                )
831
832        return self
